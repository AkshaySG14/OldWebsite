    <script>
        $(function() {
            // Gets the div that houses the source code.
            const activeSource = $("#active-source");
            // Initialize variable responsible for deciding source code to display.
            var source = 0;

            // Renews the PrismJS script so that the code is properly highlighted.
            function renewPrism() {
                $('script[src="Static/js/prism.js"]').remove();
                $('<script>').attr('src', "Static/js/prism.js").appendTo('head');
            }

            // Script responsible for setting the text of the source code being displayed.
            function setScript() {
                switch (parseInt(source)) {
                    case 0:
                        // Sends an async request to load the relevant source code, and then reinitializes the
                        // highlighting script (prism.js).
                        activeSource.load('Static/SourceCode/Website/HomePage.html', renewPrism);
                        break;
                    case 1:
                        activeSource.load('Static/SourceCode/Website/Memori.html', renewPrism);
                        break;
                    case 2:
                        activeSource.load('Static/SourceCode/Website/MagneticMotionDemo.html', renewPrism);
                        break;
                    case 3:
                        activeSource.load('Static/SourceCode/Website/PixelKnight.html', renewPrism);
                        break;
                    case 4:
                        activeSource.load('Static/SourceCode/Website/DarkAndLight.html', renewPrism);
                        break;
                    case 5:
                        activeSource.load('Static/SourceCode/Website/FatesReprise.html', renewPrism);
                        break;
                    case 6:
                        activeSource.load('Static/SourceCode/Website/ColorSynthesizer.html', renewPrism);
                        break;
                    case 7:
                        activeSource.load('Static/SourceCode/Website/DocumentEncryptor.html', renewPrism);
                        break;
                    case 8:
                        activeSource.load('Static/SourceCode/Website/FileEncryptor.html', renewPrism);
                        break;
                    case 9:
                        activeSource.load('Static/SourceCode/Website/TrafficDirector.html', renewPrism);
                        break;
                    case 10:
                        activeSource.load('Static/SourceCode/Website/SpaceShooter.html', renewPrism);
                        break;

                }
            }

            // Clears all the HTML of the source code div.
            function clearScript() {
                if (activeSource) {
                    activeSource.html("");
                }
            }

            // When clicking a specific source code to view in the dropdown bar.
            $(".dropdown-item").click(function () {
                $('#dropdownMenuButton').text($(this).text());
                source = $(this).attr('data-val');
                clearScript();
                setScript();
            });
            // Sets initial source code.
            setScript();
        });
        </script>
